-- Q1: List all customers from Mumbai along with their total order value
SELECT customer_id, SUM(total_amount)
FROM orders
GROUP BY customer_id;

-- Q2: Find top 3 products by total quantity sold
SELECT product_id, SUM(quantity) AS total_quantity
FROM order_items
GROUP BY product_id
ORDER BY total_quantity DESC
LIMIT 3;

-- Q3: Sales representatives and number of unique customers handled
SELECT rep_id, COUNT(DISTINCT customer_id)
FROM orders
GROUP BY rep_id;

-- Q4: Orders where value exceeds 10000
SELECT order_id,total_amount
FROM orders
WHERE total_amount > 10000
ORDER BY total_amount DESC;

-- Q5: Products never ordered
SELECT product_name
FROM products
WHERE product_id NOT IN
(SELECT product_id FROM order_items);
